})

io.instruc:=0.U

/*

	}

	io.instruc := "b000000000000_00000_000_00000_0010011".U // NOP

	io.instruc := "b000000000000_00000_000_00000_0010011".U // NOP
	}

	io.instruc := "b000000000001_00000_000_00001_0010011".U
	}

	}

	io.instruc := "b000000000101_00000_000_00001_0010011".U
